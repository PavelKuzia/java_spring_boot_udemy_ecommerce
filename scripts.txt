# start a PostgreSQL container
docker run --name postgres_db -e POSTGRES_PASSWORD=ecommerce -p 5432:5432 -v db:/var/lib/postgresql/data -d postgres

# enter SQL command
docker exec -it -u postgres postgres_db psql

# run pgadmin
docker run --name pgadmin-container -p 5050:80 -e PGADMIN_DEFAULT_EMAIL=pavelkuznec132@gmail.com -e PGADMIN_DEFAULT_PASSWORD=secret -d dpage/pgadmin4

# get ip of the postgres container
docker inspect -f "{{range .NetworkSettings.Networks}}{{.IPAddress}}{{end}}" postgres_db
